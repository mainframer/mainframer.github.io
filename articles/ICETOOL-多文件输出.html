ICETOOL: 多文件输出

2012-08-02

取一个文件MYINPUT中的有用字段，组装成多个不同LRECL的输出文件。这个也是项目中用到的，我再这里简化了好多！

//STEP001 EXEC PGM=ICETOOL,REGION=4096K,COND=(0,NE)         
//TOOLMSG  DD SYSOUT=*                                      
//DFSMSG   DD SYSOUT=*                                      
//SYSOUT   DD SYSOUT=*                                      
//SYSPRINT DD SYSOUT=*                                      
//SORTWK01 DD UNIT=SYSDA,SPACE=(CYL,(30,3),RLSE)            
//IN      DD DSN=MYINPUT,DISP=SHR :LRECL=05                 
//OUT1    DD DSN=MYOUT1,DISP=SHR  :LRECL=10                 
//OUT2    DD DSN=MYOUT2,DISP=SHR  :LRECL=20                 
//TOOLIN  DD *                                              
  COPY FROM(IN) TO(OUT1)    USING(CTL1)                     
  COPY FROM(IN) TO(OUT2)    USING(CTL2)
